// case: calculator
routine main() is
    var a: integer;
    var b: integer;
    var op: string;
    var result: integer;

    read(a);
    read(b);
    read(op);

    if op = "+" then
        result := a + b;
    end;
    if op = "-" then
        result := a - b;
    end;
    if op = "*" then
        result := a * b;
    end;
    if op = "/" then
        result := a / b;
    end;

    print(result);
end;
